<script>

const PIN="9747";

const DAILY_GOAL=15;

const WEEKLY_GOAL=50;

const kids=[

{name:"Adam",daily:0,total:0,weekly:0,treatsToday:0},

{name:"Lachlan",daily:0,total:0,weekly:0,treatsToday:0}

];

const chores={

Morning:[["🛏️","Make bed",2],["👕","Get dressed",1],["🧼","Wash face",1],["🎒","Pack school bag",1]],

Afternoon:[["🎒","Unpack school bag",1],["📚","Do homework",2],["🍽️","Set table",2]],

Evening:[["🍽️","Clean table",1],["🚿","Shower & brush teeth",1],["🧸","Pack away toys",2]]

};

const extras=[["🍽️","Unload dishwasher",3],["🧺","Empty clothes baskets",3]];

function isWeekend(){

const d=new Date().getDay();

return d===5||d===6||d===0;

}

function render(i){

let c=kids[i];

let el=document.getElementById("child"+i);

el.innerHTML='<h2>'+c.name+'</h2>';

for(let t in chores){

el.innerHTML+='<div class="header '+t.toLowerCase()+'">'+t+'</div>';

chores[t].forEach(ch=>{

el.innerHTML+=`

<div class="chore">

<span>${ch[0]} ${ch[1]} <span class="value">⭐${ch[2]}</span></span>

<button onclick="done(${i},${ch[2]})">✓</button>

</div>`;

});

}

let ex=extras[i];

el.innerHTML+=`

<div class="chore">

<span>${ex[0]} ${ex[1]} <span class="value">⭐${ex[2]}</span></span>

<button onclick="done(${i},${ex[2]})">✓</button>

</div>`;

el.innerHTML+=`

<div class="summary">

⭐ Today: ${c.daily}<br>

⭐ Total: ${c.total}<br>

⭐ This Week: ${c.weekly}

<div class="progress">

<div class="progress-fill" style="width:${Math.min(c.weekly/WEEKLY_GOAL*100,100)}%;background:#66cc99"></div>

</div>

</div>

<div class="shop">

<b>🎁 Reward Shop</b>

<button onclick="redeemTreat(${i})">🍪 Treat – ⭐5 (max 2/day)</button>

<button onclick="redeemYouTube(${i})">📺 YouTube Time – ⭐40 (Fri–Sun)</button>

<button onclick="redeemWeekly(${i})">🎥 Movie Night – ⭐50 (Fri–Sun)</button>

<button onclick="redeemWeekly(${i})">🎲 Board Game Night – ⭐50 (Fri–Sun)</button>

</div>

<div class="parent">

<button onclick="parentReset(${i})">🔒 Parent Reset</button>

</div>`;

}

function done(i,s){

kids[i].daily+=s;

kids[i].total+=s;

kids[i].weekly+=s;

update();

if(kids[i].daily>=DAILY_GOAL)win(i);

}

function redeemTreat(i){

if(kids[i].treatsToday>=2) return alert("Treat limit reached today");

if(kids[i].total<5) return alert("Not enough stars");

let pin=prompt("Enter parent PIN:");

if(pin===PIN){

kids[i].total-=5;

kids[i].treatsToday++;

update();

}else alert("Incorrect PIN");

}

function redeemYouTube(i){

if(!isWeekend()) return alert("YouTube time is only available Fri–Sun");

if(kids[i].total<40) return alert("Not enough stars");

let pin=prompt("Enter parent PIN:");

if(pin===PIN){

kids[i].total-=40;

update();

}else alert("Incorrect PIN");

}

function redeemWeekly(i){

if(!isWeekend()) return alert("Weekly rewards are only available Fri–Sun");

if(kids[i].weekly<50) return alert("Keep going!");

let pin=prompt("Enter parent PIN:");

if(pin===PIN){

kids[i].weekly-=50;

update();

}else alert("Incorrect PIN");

}

function parentReset(i){

let pin=prompt("Enter parent PIN:");

if(pin===PIN){

kids[i].daily=0;kids[i].total=0;kids[i].weekly=0;kids[i].treatsToday=0;

update();

}else alert("Incorrect PIN");

}

function update(){

kids.forEach((k,i)=>{

document.getElementById("race"+i).style.width=Math.min(k.daily/DAILY_GOAL*100,100)+'%';

render(i);

});

localStorage.setItem("kids",JSON.stringify(kids));

}

function win(i){

let w=document.getElementById("winner");

w.style.display="flex";

w.textContent="🎉 "+kids[i].name+" Wins Today! 🎉";

setTimeout(()=>w.style.display="none",4000);

}

function autoReset(){

let now=new Date();

let today=now.toDateString();

let lastDay=localStorage.getItem("lastDay");

if(lastDay!==today){

kids.forEach(k=>{k.daily=0;k.treatsToday=0});

localStorage.setItem("lastDay",today);

}

let day=now.getDay();

let lastWeek=localStorage.getItem("lastWeekReset");

if(day===1 && lastWeek!==today){

kids.forEach(k=>k.weekly=0);

localStorage.setItem("lastWeekReset",today);

}

localStorage.setItem("kids",JSON.stringify(kids));

}

let saved=localStorage.getItem("kids");

if(saved)kids.splice(0,kids.length,...JSON.parse(saved));

autoReset();

setInterval(autoReset,60000);

update();

</script>
